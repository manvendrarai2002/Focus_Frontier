💻 How It's Triggered in Your Games

// In any game (e.g., memory-matrix-ml.js)
class MemoryMatrixGame extends GameEngine {
  
  completeGame() {
    this.isGameActive = false;
    
    // Calculate final metrics
    this.gameMetrics.score = this.calculateScore();
    this.gameMetrics.accuracy = this.matches / (this.matches + this.misses);
    this.gameMetrics.moves = this.matches + this.misses;
    
    // THIS TRIGGERS THE RESULTS DISPLAY! 👇
    this.endGame(); // ← GameEngine method
    
    // endGame() automatically:
    // 1. Saves metrics to database
    // 2. Calls ML analysis API
    // 3. Displays results with showGameResults()
  }
}
